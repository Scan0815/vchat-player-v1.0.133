import { expect } from 'chai';
import { TargetSet } from 'vchat-core';
import { Player } from '../src/player';
import { PlayerConfig } from '../src/player-config';
 
describe('Player', () => {
    beforeEach(() => {
        const playerConfig: PlayerConfig = {
            container: document.createElement('div'),
            factories: [],
            handler: null
        };
 
        this.objectUnderTest = new Player(playerConfig);
    });
 
    it('publish with an empty targetSet returns false', () => {
        const targetSet: TargetSet = {};
 
        const result = this.objectUnderTest.publish(targetSet);
 
        expect(result).is.false;
    });
 
    describe('startCamera', () => {
        it('does call onCameraOn handler', (done) => {
            const playerConfig: PlayerConfig = {
                container: document.createElement('div'),
                factories: [],
                handler: {
                    onCameraOn: () => { done(); }
                }
            };
 
            const objectUnderTest = new Player(playerConfig);
            objectUnderTest.startCamera();
        });
    });
 
    describe('to jpeg target', () => {
        it('with no started camera must throw an error', () => {
            const targetSet: TargetSet = { jpeg: [{ stream: 'any stream' }] };
 
            expect(() => this.objectUnderTest.publish(targetSet)).to.throw();
        });

    });
});
