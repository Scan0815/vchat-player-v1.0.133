import { TargetSet, PicUpStream, SourceSet, Source } from 'vchat-core';
import { Player } from '../src/player';
import { PlayerConfig } from '../src/player-config';
import { PlayerHandler } from '../src/player-handler';
import 'jasmine';
import { PlayerFactory } from '../src/player-factory';
import { mock, instance, when, anything, verify, capture, anyOfClass } from 'ts-mockito';
import { IPlayer } from '../src/iplayer';
 
describe('Player', () => {
    let objectUnderTest: Player;
    let handlerSpy: jasmine.SpyObj<PlayerHandler>;
 
    let mockedPlayerOne: IPlayer;
    let mockedPlayerOneFactory: PlayerFactory;
    let playerOneFactory: PlayerFactory;
 
    let mockedPlayerTwo: IPlayer;
    let mockedPlayerTwoFactory: PlayerFactory;
 
    beforeEach(() => {
        handlerSpy = jasmine.createSpyObj<PlayerHandler>([
            'onVolumeChange',
            'onPlayerSwitch',
            'onPlayerSwitchComplete',
            'onError',
            'onSendMetrics',
            'onPlayStart',
            'onPlayStop',
            'onPlayInfo',
            'onPlayError',
            'onCameraOn',
            'onCameraOff',
            'onCameraDenied',
            'onPublishStart',
            'onPublishStop'
        ]);
 
        mockedPlayerOne = mock<IPlayer>();
        const playerOne = instance(mockedPlayerOne);
        mockedPlayerOneFactory = mock<PlayerFactory>();
        playerOneFactory = instance(mockedPlayerOneFactory);
        when(mockedPlayerOneFactory.format).thenReturn(['hls']);
        when(mockedPlayerOneFactory.isSupported()).thenReturn(true);
        when(mockedPlayerOneFactory.create(anything())).thenResolve(playerOne);
 
 
        mockedPlayerTwo = mock<IPlayer>();
        const playerTwo = instance(mockedPlayerTwo);
        mockedPlayerTwoFactory = mock<PlayerFactory>();
        const playerTwoFactory = instance(mockedPlayerTwoFactory);
        when(mockedPlayerTwoFactory.format).thenReturn(['hls']);
        when(mockedPlayerTwoFactory.isSupported()).thenReturn(true);
        when(mockedPlayerTwoFactory.create(anything())).thenResolve(playerTwo);
 
        const playerConfig: PlayerConfig = {
            container: document.createElement('div'),
            factories: [playerOneFactory, playerTwoFactory],
            handler: handlerSpy
        };
 
        objectUnderTest = new Player(playerConfig);
    });
 
    describe('play', () => {
        it('does something', () => {
            const sourceSet: SourceSet = {
                hls: [{stream: 'test_stream_url'}]
            };
 
            console.log('PLAYING');
            objectUnderTest.play(sourceSet);
            console.log('DONE');
            verify(mockedPlayerOneFactory.isSupported()).once();
            verify(mockedPlayerTwoFactory.isSupported()).once();
 
            verify(mockedPlayerOneFactory.create(anything(), anything())).once();
            verify(mockedPlayerTwoFactory.create(anything(), anything())).never();
 
        });
    });
 
    describe('hasCamera', () => {
        const createMediaDeviceInfo: (
            kind: MediaDeviceKind
        ) => MediaDeviceInfo = (kind: MediaDeviceKind) => {
            return {
                deviceId: '123',
                groupId: '456',
                kind: kind,
                label: 'test_device',
                toJSON: (): string => ''
            };
        };
 
        it('with no camera returns false', async () => {
            const mediaDeviceInfos: MediaDeviceInfo[] = [
                createMediaDeviceInfo('audioinput'),
                createMediaDeviceInfo('audiooutput')
            ];
 
            spyOn(navigator.mediaDevices, 'enumerateDevices').and.returnValue(
                Promise.resolve<MediaDeviceInfo[]>(mediaDeviceInfos)
            );
 
            const hasCameraResult = await objectUnderTest.hasCamera();
 
            expect(hasCameraResult).toBe(false);
        });
 
        it('with one or more camera returns true', async () => {
            const mediaDeviceInfos: MediaDeviceInfo[] = [
                createMediaDeviceInfo('audioinput'),
                createMediaDeviceInfo('videoinput')
            ];
 
            spyOn(navigator.mediaDevices, 'enumerateDevices').and.returnValue(
                Promise.resolve<MediaDeviceInfo[]>(mediaDeviceInfos)
            );
 
            const hasCameraResult = await objectUnderTest.hasCamera();
 
            expect(hasCameraResult).toBe(true);
        });
    });
 
    describe('startCamera', () => {
        it('with the camera being denied calls camera denied handler', async () => {
            spyOn<any>(objectUnderTest, '_getUserMedia').and.returnValue(
                Promise.reject(
                    new DOMException('not allowed', 'NotAllowedError')
                )
            );
 
            await objectUnderTest.startCamera();
 
            expect(handlerSpy.onCameraDenied).toHaveBeenCalled();
            expect(handlerSpy.onCameraOff).not.toHaveBeenCalled();
        });
 
        it('with no camera available calls camera denied handler', async () => {
            spyOn<any>(objectUnderTest, '_getUserMedia').and.returnValue(
                Promise.reject(new DOMException('not found', 'NotFoundError'))
            );
 
            await objectUnderTest.startCamera();
 
            expect(handlerSpy.onCameraOff).toHaveBeenCalled();
            expect(handlerSpy.onCameraDenied).not.toHaveBeenCalled();
        });
 
        it('does call onCameraOn handler', (done) => {
            const playerConfig: PlayerConfig = {
                container: document.createElement('div'),
                factories: [],
                handler: {
                    onCameraOn: (): void => {
                        done();
                    }
                }
            };
 
            const objectUnderTest = new Player(playerConfig);
            objectUnderTest.startCamera();
        });
    });
 
    describe('publish', () => {
        it('with an empty targetSet returns false', () => {
            const targetSet: TargetSet = {};
 
            const result = objectUnderTest.publish(targetSet);
 
            expect(result).toBe(false);
        });
 
        describe('called twice while still publishing', () => {
            const targetSet: TargetSet = {
                jpeg: [
                    {
                        stream: 'any stream',
                        width: 640,
                        height: 480,
                        fps: 20,
                        quality: 1
                    }
                ]
            };
 
            let dummyCamPublisher: jasmine.SpyObj<PicUpStream>;
            let playerCreateCamPublisherSpy: jasmine.Spy;
 
            beforeEach(async () => {
                await objectUnderTest.startCamera();
 
                dummyCamPublisher = jasmine.createSpyObj<PicUpStream>([
                    'publish'
                ]);
                playerCreateCamPublisherSpy = spyOn<any>(
                    objectUnderTest,
                    '_createCamPublisher'
                ).and.returnValue(dummyCamPublisher);
 
                objectUnderTest.publish(targetSet);
 
                playerCreateCamPublisherSpy.calls.reset();
                dummyCamPublisher.publish.calls.reset();
            });
 
            it('is ignored if using an equal target set', () => {
                const publishResult = objectUnderTest.publish(targetSet);
 
                expect(publishResult).toBe(true);
                expect(playerCreateCamPublisherSpy).not.toHaveBeenCalled();
                expect(dummyCamPublisher.publish).not.toHaveBeenCalled();
            });
 
            it('throws if using a different target set', () => {
                const differentTargetSet: TargetSet = {
                    jpeg: [
                        {
                            stream: 'other stream',
                            width: 1024,
                            height: 768,
                            fps: 20,
                            quality: 1
                        }
                    ]
                };
 
                expect(() =>
                    objectUnderTest.publish(differentTargetSet)
                ).toThrowError();
            });
        });
 
        describe('to jpeg target', () => {
            it('with no started camera must throw an error', () => {
                const targetSet: TargetSet = {
                    jpeg: [
                        {
                            stream: 'any stream',
                            width: 640,
                            height: 480,
                            fps: 20,
                            // eslint-disable-next-line unicorn/no-zero-fractions
                            quality: 1.0
                        }
                    ]
                };
 
                expect(() => objectUnderTest.publish(targetSet)).toThrow();
            });
        });
    });
 
    describe('constructor', () => {
        it('throws if handler is null', () => {
            const playerConfig: PlayerConfig = {
                container: document.createElement('div'),
                factories: [],
                handler: null
            };
 
            expect(() => new Player(playerConfig)).toThrowError();
        });
    });
 
    describe('initPlayer', () => {
        it('with single format', () => {
            objectUnderTest.initPlayer('test_format_one');
        });
 
        it('with multiple formats', () => {
            objectUnderTest.initPlayer(['test_format_one', 'test_format_two']);
        });
    });
});
